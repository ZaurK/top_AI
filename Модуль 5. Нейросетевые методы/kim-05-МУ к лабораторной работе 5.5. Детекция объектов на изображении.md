### 2. Теоретические сведения

#### 2.1. Задача детекции объектов

Детекция объектов — задача компьютерного зрения, заключающаяся в определении наличия объектов на изображении и их локализации с помощью ограничивающих рамок (bounding boxes).

**Формат аннотации**:
- Координаты центра \( (cx, cy) \), ширина \( w \), высота \( h \);
- Или координаты углов \( (x_1, y_1, x_2, y_2) \).

#### 2.2. Архитектуры детекторов

**YOLO (You Only Look Once)**:
- Единая свёрточная сеть, предсказывающая bounding boxes и классы;
- Разбивает изображение на сетку \( S \times S \);
- Каждая ячейка предсказывает \( B \) bounding boxes и вероятности классов;
- Очень высокая скорость работы (до 155 FPS).

**Faster R-CNN**:
- Двухэтапный подход:
  1. **Region Proposal Network (RPN)** — предлагает регионы с объектами;
  2. **Классификатор** — классифицирует регионы и уточняет координаты;
- Высокая точность, но медленная скорость.

**SSD (Single Shot Detector)**:
- Одновременное предсказание на нескольких масштабах;
- Использует якорные прямоугольники (anchor boxes);
- Хороший баланс между скоростью и точностью.

#### 2.3. Метрики качества детекции

**IoU (Intersection over Union)**:
\[
IoU = \frac{Area(B_{pred} \cap B_{gt})}{Area(B_{pred} \cup B_{gt})}
\]

**AP (Average Precision)**:
- Площадь под кривой Precision-Recall.

**mAP (mean Average Precision)**:
- Среднее AP по всем классам.

2. Преобразование в формат COCO или YOLO:
   ```python
   # Формат YOLO: [class, x_center_norm, y_center_norm, width_norm, height_norm]
   def yolo_format(bbox, img_width, img_height):
       x1, y1, x2, y2 = bbox
       x_center = (x1 + x2) / 2 / img_width
       y_center = (y1 + y2) / 2 / img_height
       width = (x2 - x1) / img_width
       height = (y2 - y1) / img_height
       return [x_center, y_center, width, height]
   ```

#### 3.2. Использование предобученной модели YOLO

```python
# Установка Ultralytics YOLO

from ultralytics import YOLO

# Загрузка предобученной модели
model = YOLO('yolov8n.pt')  # YOLOv8 nano

# Функция для детекции
def detect_objects(image_path, confidence=0.5, iou=0.45):
    results = model(image_path, conf=confidence, iou=iou)
    
    boxes = []
    labels = []
    scores = []
    
    for r in results:
        for box in r.boxes:
            x1, y1, x2, y2 = box.xyxy[0].tolist()
            conf = box.conf[0].tolist()
            cls = int(box.cls[0].tolist())
            
            boxes.append([int(x1), int(y1), int(x2), int(y2)])
            labels.append(model.names[cls])
            scores.append(conf)
    
    return boxes, labels, scores

# Визуализация результатов
def visualize_detection(image_path, boxes, labels, scores):
    img = cv2.imread(image_path)
    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    
    colors = np.random.randint(0, 255, (len(set(labels)), 3))
    
    for box, label, score in zip(boxes, labels, scores):
        x1, y1, x2, y2 = box
        color = colors[hash(label) % len(colors)]
        cv2.rectangle(img, (x1, y1), (x2, y2), color.tolist(), 2)
        cv2.putText(img, f'{label}: {score:.2f}', (x1, y1-5),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, color.tolist(), 2)
    
    plt.figure(figsize=(12, 8))
    plt.imshow(img)
    plt.axis('off')
    plt.show()
```

#### 3.3. Обучение кастомной модели YOLO

```python
# Подготовка датасета в формате YOLO
def prepare_yolo_dataset(images_dir, annotations, output_dir='dataset'):
    os.makedirs(output_dir, exist_ok=True)
    os.makedirs(f'{output_dir}/images/train', exist_ok=True)
    os.makedirs(f'{output_dir}/labels/train', exist_ok=True)
    os.makedirs(f'{output_dir}/images/val', exist_ok=True)
    os.makedirs(f'{output_dir}/labels/val', exist_ok=True)
    
    # Создание файла с классами
    classes = list(set(ann['label'] for ann in annotations))
    with open(f'{output_dir}/classes.txt', 'w') as f:
        for cls in classes:
            f.write(f'{cls}\n')
    
    # Разделение на train/val
    train_ann, val_ann = train_test_split(annotations, test_size=0.2, random_state=42)
    
    # Создание YOLO аннотаций
    for ann in train_ann:
        img = cv2.imread(ann['image_path'])
        h, w = img.shape[:2]
        label_path = ann['image_path'].replace('images', 'labels').replace('.jpg', '.txt')
        with open(label_path, 'w') as f:
            for bbox, label in zip(ann['bboxes'], ann['labels']):
                yolo_bbox = yolo_format(bbox, w, h)
                cls_id = classes.index(label)
                f.write(f"{cls_id} {yolo_bbox[0]} {yolo_bbox[1]} {yolo_bbox[2]} {yolo_bbox[3]}\n")
```

#### 3.4. Использование Faster R-CNN

```python
import torch
import torchvision
from torchvision.models.detection import fasterrcnn_resnet50_fpn
from torchvision.transforms import functional as F

# Загрузка предобученной модели
model = fasterrcnn_resnet50_fpn(pretrained=True)
model.eval()

def detect_with_fasterrcnn(image_path, confidence_threshold=0.5):
    # Загрузка и предобработка изображения
    image = cv2.imread(image_path)
    image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    image_tensor = F.to_tensor(image).unsqueeze(0)
    
    # Инференс
    with torch.no_grad():
        predictions = model(image_tensor)
    
    # Обработка результатов
    boxes = predictions[0]['boxes'].numpy()
    scores = predictions[0]['scores'].numpy()
    labels = predictions[0]['labels'].numpy()
    
    # Фильтрация по порогу уверенности
    mask = scores >= confidence_threshold
    boxes = boxes[mask].astype(int)
    scores = scores[mask]
    labels = labels[mask]
    
    return boxes, labels, scores
```

#### 3.5. Исследование параметров детекции

```python
def analyze_parameters(image_path, confidence_range, nms_range):
    results = []
    
    for conf in confidence_range:
        for iou in nms_range:
            boxes, labels, scores = detect_objects(
                image_path,
                confidence=conf,
                iou=iou
            )
            results.append({
                'confidence': conf,
                'iou': iou,
                'num_objects': len(boxes),
                'avg_score': np.mean(scores) if scores else 0
            })
    
    return results

# Визуализация влияния параметров
def plot_parameter_analysis(results):
    fig, axes = plt.subplots(1, 2, figsize=(12, 5))
    
    conf_values = sorted(set(r['confidence'] for r in results))
    iou_values = sorted(set(r['iou'] for r in results))
    
    # Количество объектов
    data = np.array([[r['num_objects'] for r in results if r['confidence'] == c]
                     for c in conf_values])
    axes[0].imshow(data, cmap='Blues')
    axes[0].set_xticks(range(len(iou_values)))
    axes[0].set_xticklabels(iou_values)
    axes[0].set_yticks(range(len(conf_values)))
    axes[0].set_yticklabels(conf_values)
    axes[0].set_xlabel('IoU Threshold')
    axes[0].set_ylabel('Confidence Threshold')
    axes[0].set_title('Number of Detected Objects')
    
    # Средняя уверенность
    data = np.array([[r['avg_score'] for r in results if r['confidence'] == c]
                     for c in conf_values])
    axes[1].imshow(data, cmap='Reds')
    axes[1].set_xticks(range(len(iou_values)))
    axes[1].set_xticklabels(iou_values)
    axes[1].set_yticks(range(len(conf_values)))
    axes[1].set_yticklabels(conf_values)
    axes[1].set_xlabel('IoU Threshold')
    axes[1].set_ylabel('Confidence Threshold')
    axes[1].set_title('Average Confidence Score')
    
    plt.tight_layout()
    plt.show()
```

```python
# Функция для пакетной обработки
def batch_detect(image_dir, output_dir, model, confidence=0.5):
    os.makedirs(output_dir, exist_ok=True)
    results = []
    
    for img_file in os.listdir(image_dir):
        if img_file.endswith(('.jpg', '.png')):
            img_path = os.path.join(image_dir, img_file)
            boxes, labels, scores = detect_objects(model, img_path, confidence)
            
            # Сохранение результатов
            result = {
                'image': img_file,
                'num_objects': len(boxes),
                'labels': labels,
                'scores': scores,
                'boxes': boxes
            }
            results.append(result)
            
            # Визуализация
            visualize_detection(img_path, boxes, labels, scores)
            plt.savefig(os.path.join(output_dir, f'result_{img_file}'))
    
    return results
```
